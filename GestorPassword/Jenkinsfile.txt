        stage('Build') {
            steps {
                echo 'Compilando mi proyecto...'
                // Aquí irían comandos como 'sh "mvn compile"'
            }
        }
        stage('Test') {
            steps {
                echo 'Ejecutando pruebas...'
            }
        }
    }
}
